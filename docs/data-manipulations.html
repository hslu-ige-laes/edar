df.long <- as.data.frame(tidyr::pivot_longer(df,
                                      cols = -time,
                                      names_to = "name",
                                      values_to = "value",
                                      values_drop_na = TRUE)
                         )

# long format

# convert long table into wide table
df.wide <- as.data.frame(tidyr::pivot_wider(df.long,
                                     names_from = "name",
                                     values_from = "value")
                         )
