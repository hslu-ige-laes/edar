5.2.1 Scatterplot

5.2.1.1 plot()

# load data set
df <- read.csv("https://github.com/retomarek/r/raw/master/datasets/buildingMonitoringTestDataSet.csv",
               stringsAsFactors=FALSE,
               sep =",")

# crate simple scatterplot
plot(df$WthStnTemp, df$BldgEnergyHeating)
